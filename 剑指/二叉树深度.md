
首先是递归解法：  
```
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if(!root)  return 0;
        return max(maxDepth(root->left),maxDepth(root->right)) + 1;
    }
};
```
但可能会不让用递归，则层序遍历解法：  
结合学过的，用个队列，然后一层层放，每层将layer++，记录最大层数即可。  
```
class Solution {
public:
    int maxDepth(TreeNode* root) {
        queue<TreeNode*> Q;
        int layer = 0;
        if(!root)  return 0;
        Q.push(root);
        while(!Q.empty())
        {
            int n = Q.size();
            for(int i = 0; i < n;i++)
            {
                TreeNode* node = Q.front();
                Q.pop();
                if(node->left) Q.push(node->left);
                if(node->right) Q.push(node->right);
            }
            layer++; 
        }
        return layer;
    }
};
```
