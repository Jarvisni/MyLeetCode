给你一个整数数组 nums ，数组中的元素互不相同。返回该数组所有可能的子集（幂集）。
解集不能包含重复的子集。你可以按任意顺序返回解集
此类题目可以使用**回溯法**

```C++
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        int index=0;
        vector<int> temp;
        vector<vector<int>> ans;
        ans.push_back(temp);//注意这里要把空集这个边界放进去，这样写或者ans.push_back({});
        dfs(index,nums,temp,ans);
        return ans;
    }
private:
    void dfs(int index,vector<int>&nums,vector<int>&temp,vector<vector<int>>& ans)
    {
        if(index==nums.size())  return;
        temp.push_back(nums[index]);
        ans.push_back(temp);
        dfs(index+1,nums,temp,ans);
        temp.pop_back();
        dfs(index+1,nums,temp,ans);
    }

};
```

全排列：  
给定一个没有重复数字的序列，返回其所有可能的全排列。  
```C++
class Solution {
public:
    vector<vector<int>> ans;
    vector<int> temp;
    vector<vector<int>> permute(vector<int>& nums) {
        vector<bool> used(nums.size(), false);
        dfs(nums, used);
        return ans;
    }
    void dfs (vector<int>& nums, vector<bool>& used) {
        if (temp.size() == nums.size()) {
            ans.push_back(temp);
            return;
        }
        for (int i = 0; i < nums.size(); i++) {
            if (used[i] == true) continue;
            used[i] = true;
            temp.push_back(nums[i]);
            dfs(nums, used);
            temp.pop_back();
            used[i] = false;
        }
    }
};
```

给定一个**可包含重复数字**的序列 nums ，按任意顺序 返回所有不重复的全排列。  
```C++
class Solution {
public:
    vector<int> temp;
    vector<vector<int>> ans;
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        vector<bool> visited(nums.size(),false);
        dfs(nums,0,visited);
        return ans;
    }
    void dfs(vector<int>& nums,int n,vector<bool>& visited)
    {
        if(n==nums.size()) {ans.push_back(temp);  return;}
        
        for(int i=0;i<nums.size();i++)
            if(visited[i]==false)
            {
                if(i>0 && nums[i]==nums[i-1] && visited[i-1]!=false) continue;
                temp.push_back(nums[i]);
                visited[i]=true;
                dfs(nums,n+1,visited);
                visited[i]=false;
                temp.pop_back();
            }
    }
    
};
```
