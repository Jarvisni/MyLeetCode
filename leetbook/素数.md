题目：统计小于非负整数n的质数的数量，n<=5*10^6  
埃氏筛法：
```
class Solution {
public:
    int countPrimes(int n) {
        int ans=0;
        vector<int> vec(n,1);//1表示素，0表示为之前某数的倍数
        for(int i=2;i<n;++i)
            if(vec[i])
            {
                ans++;
                for(long long j=(long long)i*i;j<n;j+=i)
                    vec[j]=0;
            }
        return ans;
    }
};
```
